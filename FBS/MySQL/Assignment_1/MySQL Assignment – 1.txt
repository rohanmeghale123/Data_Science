min_salary, max_salary.

- job_id must be Primary Key
- job_title cannot be NULL
- max_salary must not exceed 25000

->create table jobs(job_id int primary key,job_title varchar(20) not null,min_salary decimal(7,2),max_salary decimal(7,2) check(max_salary<=25000));
